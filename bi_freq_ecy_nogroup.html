def cv(train_df, folds, alpha):
    loss = 0
    for n, (trn_idx, val_idx) in enumerate(folds.split(train_df)):
        trn_df, val_df = train_df.iloc[trn_idx], train_df.iloc[val_idx]

        preds = result.predict(val_df, exposure=val_df['ecy'])
        loss += Poisson_loss(val_df['bi_cnt'], preds)/folds.n_splits
    return preds, loss
